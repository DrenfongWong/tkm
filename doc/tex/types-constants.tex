\subsubsection{result\_type constants}

\begin{constantdefinition}{result\_type constants}
OK & \hex{0000000000000000} & Request was processed successfully \\
Invalid\_Operation & \hex{0000000000000101} & The requested operation is invalid \\
Invalid\_ID & \hex{0000000000000102} & The given identifier is invalid \\
Invalid\_State & \hex{0000000000000103} & TKM is in an invalid state to process the given request \\
Invalid\_Parameter & \hex{0000000000000104} & Invalid value given as request parameter \\
Random\_Failure & \hex{0000000000000201} & The random number generator is inoperable \\
Sign\_Failure & \hex{0000000000000202} & Signature could not be generated \\

\end{constantdefinition}
